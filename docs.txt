Install and connect to mysql:
sudo mysql -u pi -p
USE mysql;
CREATE USER 'pi'@'localhost' IDENTIFIED WITH mysql_native_password BY 'YOUR PASSWORD';
GRANT ALL PRIVILEGES ON *.* TO 'pi'@'localhost';
FLUSH PRIVILEGES;

CREATE DATABASE farm;
CREATE TABLE schedule(id INT PRIMARY KEY AUTO_INCREMENT, at TIME NOT NULL, enabled BOOLEAN NOT NULL);
CREATE TABLE logs(id INT PRIMARY KEY AUTO_INCREMENT, message VARCHAR(1024) NOT NULL, at DATETIME NOT NULL, level TINYINT UNSIGNED NOT NULL);
CREATE TABLE readings(id INT PRIMARY KEY AUTO_INCREMENT, humidity decimal(3,2) NOT NULL, temperature decimal(3, 2) NOT NULL, at DATETIME NOT NULL);

Create systemd service file in /etc/systemd called farm.service with content:

[Unit]
Description=Farm systemd service
After=mysql.service

[Service]
Type=simple
Environment="pass={your mysql user password}
ExecStart=/bin/python3 /home/pi/farm.py

[Install]
WantedBy=multi-user.target

Then execute commands:
sudo chmod 644 /etc/systemd/system/farm.service
sudo systemctl enable farm.service
sudo systemctl start farm.service
