CREATE DATABASE farm;
CREATE TABLE times(id INT PRIMARY KEY AUTO_INCREMENT, at TIME NOT NULL, enabled BOOLEAN NOT NULL);
CREATE TABLE logs(id INT PRIMARY KEY AUTO_INCREMENT, message VARCHAR(1024) NOT NULL, at DATETIME NOT NULL, level TINYINT UNSIGNED NOT NULL);
CREATE TABLE readings(id INT PRIMARY KEY AUTO_INCREMENT, humidity decimal(3,2) NOT NULL, temperature decimal(3, 2) NOT NULL, at DATETIME NOT NULL);
